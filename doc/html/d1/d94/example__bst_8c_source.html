93 int example_read_user_choice(int *choice)
94 {
95  char val;
96  char digits[MAX_DIGITS_IN_CHOICE + 1];
97  int idx = 0;
98  int valid = TRUE;
99 
100  /* parse input string until \n */
101  while((val = getchar()) != '\n')
102  {
103  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
104  {
105  digits[idx++] = val;
106  }
107  else
108  {
109  valid = FALSE;
110  }
111  }
112  if ((valid == TRUE) && idx != 0)
113  {
114  digits[idx] = '\0';
115  *choice = atoi(digits);
116  return(OPENNSL_E_NONE);
117  }
118  else
119  {
120  *choice = -1;
121  return(OPENNSL_E_FAIL);
122  }
123 }

140 /* Main function for BST application */
141 int main(int argc, char *argv[])
142 {
143  opennsl_error_t rc;
144  int choice;
145  int unit = SWITCH_UNIT_NUM;
146  int index;
147  opennsl_gport_t gport;
148  opennsl_port_t port;
149  opennsl_cos_queue_t cosq;
150  uint32 options = 0;
151  example_bst_counter_t id_list[MAX_COUNTERS] = {
152  {opennslBstStatIdUcast, "opennslBstStatIdUcast"},
153  {opennslBstStatIdMcast, "opennslBstStatIdMcast"},
154  {opennslBstStatIdPriGroupShared, "opennslBstStatIdPriGroupShared"},
155  {opennslBstStatIdPriGroupHeadroom, "opennslBstStatIdPriGroupHeadroom"}
156  };
157  uint64 val[MAX_COUNTERS][MAX_COSQ_COUNT] = {0};
158 
159  if ((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
160  printf("%s\n\r", example_usage);
161  return OPENNSL_E_PARAM;
162  }
163 
164  /* Initialize the system. */
165  printf("Initializing the system.\r\n");
166  rc = opennsl_driver_init();
167 
168  if (rc != OPENNSL_E_NONE) {
169  printf("\r\nFailed to initialize the system, rc = %d (%s).\r\n",
170  rc, opennsl_errmsg(rc));
171  return 0;
172  }
173 
174  /* Add ports to default vlan. */
175  printf("Adding ports to default vlan.\r\n");
176  rc = example_switch_default_vlan_config(unit);
177  if (rc != OPENNSL_E_NONE) {
178  printf("\r\nFailed to add default ports, rc = %d (%s).\r\n",
179  rc, opennsl_errmsg(rc));
180  }
181 
182  /* Enabling bst module. */
183  rc = opennsl_switch_control_set(unit, opennslSwitchBstEnable, 1);
184  if (rc != OPENNSL_E_NONE) {
185  printf("\r\nFailed to Enable bst, rc = %d (%s).\r\n",
186  rc, opennsl_errmsg(rc));
187  return 0;
188  }
189  printf("BST feature is enabled.\r\n");
190 
191 
192  while (1) {
193  printf("\nUser Menu: Select one of the following options\n");
194  printf("1. Display bst statistics of a port.\n");
195  printf("2. Clear bst statistics of a port.\n");
196  printf("9. Launch diagnostic shell\n");
197  printf("0. Quit the application.\n");
198 
199  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
200  {
201  printf("Invalid option entered. Please re-enter.\n");
202  continue;
203  }
204  switch(choice) {
205 
206  case 1:
207  {
208  printf("\r\nEnter the port number.\r\n");
209  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
210  {
211  printf("Invalid option entered. Please re-enter.\n");
212  continue;
213  }

215  if (rc != OPENNSL_E_NONE)
216  {
217  return OPENNSL_E_FAIL;
218  }
219  for (index = 0; index < MAX_COUNTERS; index++)
220  {
221  rc = (opennsl_cosq_bst_stat_sync(unit, id_list[index].bid));
222  if (rc != OPENNSL_E_NONE) {
223  printf("\r\nFailed to sync the state of port, rc = %d (%s).\r\n",
224  rc, opennsl_errmsg(rc));
225  break;
226  }
227  }
228 
229  for (index = 0; index < MAX_COUNTERS; index++)
230  {
231  for (cosq = 0; cosq <=7; cosq++)
232  {
233  rc = opennsl_cosq_bst_stat_get(unit, gport, cosq,
234  id_list[index].bid, options, &val[index][cosq]);
235  if (rc != OPENNSL_E_NONE) {
236  printf("\r\nFailed to get the port stats, rc = %d (%s).\r\n",
237  rc, opennsl_errmsg(rc));

239  }
240  printf("BST Counter: %s for COS queue: %d is : %llu\n",
241  id_list[index].name, cosq, val[index][cosq]);
242  }
243  printf("\n");
244  }
245 
246  break;
247  } /* End of case 1 */
248 
249  case 2:
250  {
251  printf("\r\nEnter the port number.\r\n");
252  if(example_read_user_choice(&port) != OPENNSL_E_NONE)
253  {
254  printf("Invalid option entered. Please re-enter.\n");
255  continue;
256  }
257  rc = opennsl_port_gport_get (unit, port, &gport);
258  if (rc!= OPENNSL_E_NONE)
259  {
260  return OPENNSL_E_FAIL;
261  }
262 
263  for (index = 0; index < MAX_COUNTERS; index++)
264  {
265  for (cosq = 0; cosq <= 7; cosq++)

273  }
274  }
275  printf("\n");
276  }
277 
278  printf("\r\nPort %d stats cleared\r\n", port);

280  } /* End of case 2 */
281 
282  case 9:

286  }
287  case 0:
288  {
289  printf("Exiting the application.\n");
290  return OPENNSL_E_NONE;
291  }
292 
293  default :
294  break;
295  } /* End of switch */
296  } /* End of while */
297 
298  return 0;
299 }
