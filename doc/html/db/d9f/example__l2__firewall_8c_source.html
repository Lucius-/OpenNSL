18  * \file example_l2_firewall.c
19  *
20  * \brief Example code for Layer2 firewall
21  *
22  * \details This example demonstrates a simple Layer 2 Firewall use case,
23  * where a network administrator can block unauthorized stations from accessing
24  * the network. The match criteria for the firewall rules can be either source
25  * MAC address or destination MAC address. Frames matching the firewall rule
26  * can either be dropped or forwarded. A centralized firewall server can
27  * maintain the firewall rules for all the Layer2 switches in the network.
28  * If the match criteria is source MAC address, packets originating with that
29  * MAC address are matched. If the match criteria is destination MAC address,
30  * packets destined to that MAC address are matched.
31  *
32  ************************************************************************/

67 int example_l2_firewall_block_station(opennsl_mac_t mac, opennsl_vlan_t vlan)
68 {
69  opennsl_error_t rv;
70  opennsl_l2_addr_t addr;
71 
72  opennsl_l2_addr_t_init(&addr, mac, vlan);
73 
74  /* Set the static flag, discard flags */
75  addr.flags = (OPENNSL_L2_STATIC | OPENNSL_L2_DISCARD_SRC |
76  OPENNSL_L2_DISCARD_DST);
77 
78  rv = opennsl_l2_addr_add(DEFAULT_UNIT, &addr);
79 
80  return rv;
81 }

92 int example_l2_firewall_allow_station(opennsl_mac_t mac, opennsl_vlan_t vlan)
93 {
94  opennsl_error_t rv;
95  opennsl_l2_addr_t addr;
96 
97  /* Check whether the station exists in the firewall rules */
98  rv = opennsl_l2_addr_get(DEFAULT_UNIT, mac, vlan, &addr);
99 
100  if (rv == OPENNSL_E_NONE) {
101  rv = opennsl_l2_addr_delete(DEFAULT_UNIT, mac, vlan);
102  }
103 
104  return rv;
105 }

147 int example_read_user_choice(int *choice)
148 {
149  char val;
150  char digits[MAX_DIGITS_IN_CHOICE + 1];
151  int idx = 0;
152  int valid = TRUE;
153 
154  /* parse input string until \n */
155  while((val = getchar()) != '\n')
156  {
157  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
158  {
159  digits[idx++] = val;
160  }
161  else
162  {
163  valid = FALSE;
164  }
165  }
166  if ((valid == TRUE) && idx != 0)
167  {
168  digits[idx] = '\0';
169  *choice = atoi(digits);
170  return(OPENNSL_E_NONE);
171  }
172  else
173  {
174  *choice = -1;
175  return(OPENNSL_E_FAIL);
176  }
177 }

187 int main(int argc, char *argv[])
188 {
189  opennsl_error_t rv;
190  opennsl_vlan_t vlan = 1;
191  int choice;
192 
193  /* Example MAC address of stations A, B, C and D */
194  opennsl_mac_t station_a = {0x00, 0x00, 0x00, 0x00, 0x00, 0x1};
195  opennsl_mac_t station_d = {0x00, 0x00, 0x00, 0x00, 0x00, 0x4};
196 
197  if((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
198  printf("%s\n\r", example_usage);
199  return OPENNSL_E_PARAM;
200  }
201 
202  /* Initialize the system. */
203  printf("Initializing the system.\r\n");
204  rv = opennsl_driver_init();
205 
206  if(rv != OPENNSL_E_NONE) {
207  printf("\r\nFailed to initialize the system. Error %s\r\n",
208  opennsl_errmsg(rv));
209  return rv;
210  }
211 
212  /* Add ports to default vlan. */
213  printf("Adding ports to default vlan.\r\n");
214  rv = example_switch_default_vlan_config(0);
215  if(rv != OPENNSL_E_NONE) {
216  printf("\r\nFailed to add default ports. rv: %d\r\n", rv);
217  return rv;
218  }
219 
220  while (1) {
221  printf("\r\nUser menu: Select one of the following options\r\n");
222  printf("1. Apply firewall\n");
223  printf("2. Remove firewall\n");
224  printf("9. Launch diagnostic shell\n");
225  printf("0. Quit the application\n");
226 
227  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
228  {
229  printf("Invalid option entered. Please re-enter.\n");
230  continue;
231  }
232  switch(choice){
233 
234  case 1:
235  {
236  /* Apply firewall rules to block station A and D */
237  rv = example_l2_firewall_block_station(station_a, vlan);
238  if (rv == OPENNSL_E_NONE) {
239  printf("L2 firewall: station A blocked successfully\n");
240  } else {
241  printf("L2 firewall: failed to apply rule. Error %s\n",
242  opennsl_errmsg(rv));
243  return rv;
244  }
245 
246  rv = example_l2_firewall_block_station(station_d, vlan);
247  if (rv == OPENNSL_E_NONE) {
248  printf("L2 firewall rule: station D blocked successfully\n");
249  } else {
250  printf("L2 firewall: failed to apply rule. Error %s\n",
251  opennsl_errmsg(rv));
252  return rv;
253  }
254  break;
255  } /* End of case 1 */
256  case 2:
257  {
258  /* Remove the firewall rules applied to station A and D */
259  rv = example_l2_firewall_allow_station(station_a, vlan);
260  if ((rv == OPENNSL_E_NONE) || (rv == OPENNSL_E_NOT_FOUND)) {
261  printf("L2 firewall: station A un-blocked successfully\n");
262  } else {
263  printf("L2 firewall: failed to apply rule. Error %s\n",
264  opennsl_errmsg(rv));
265  return rv;
266  }
267 
268  rv = example_l2_firewall_allow_station(station_d, vlan);
269  if ((rv == OPENNSL_E_NONE) || (rv == OPENNSL_E_NOT_FOUND)) {
270  printf("L2 firewall rule: station D un-blocked successfully\n");
271  } else {
272  printf("L2 firewall: failed to apply rule. Error %s\n",
273  opennsl_errmsg(rv));
274  return rv;
275  }
276  break;
277  } /* End of case 2 */
278  case 9:
279  {
280  opennsl_driver_shell();
281  break;
282  }
283  case 0:
284  {
285  printf("Exiting the application.\n");
286  return OPENNSL_E_NONE;
287  }
288  default:
289  break;
290  } /* End of switch */
291  } /* End of while */
292 
293  return OPENNSL_E_NONE;
294 }
