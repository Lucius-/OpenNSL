18  * \file example_l2_firewall.c
19  *
20  * \brief Example code for Layer2 firewall
21  *
22  * \details This example demonstrates a simple Layer 2 Firewall use case,
23  * where a network administrator can block unauthorized stations from accessing
24  * the network. The match criteria for the firewall rules can be either source
25  * MAC address or destination MAC address. Frames matching the firewall rule
26  * can either be dropped or forwarded. A centralized firewall server can
27  * maintain the firewall rules for all the Layer2 switches in the network.
28  * If the match criteria is source MAC address, packets originating with that
29  * MAC address are matched. If the match criteria is destination MAC address,
30  * packets destined to that MAC address are matched.
31  *
32  ************************************************************************/

66 int example_l2_firewall_block_station(opennsl_mac_t mac, opennsl_vlan_t vlan)
67 {
68  opennsl_error_t rv;
69  opennsl_l2_addr_t addr;
70 
71  opennsl_l2_addr_t_init(&addr, mac, vlan);
72 
73  /* Set the static flag, discard flags */
74  addr.flags = (OPENNSL_L2_STATIC | OPENNSL_L2_DISCARD_SRC |
75  OPENNSL_L2_DISCARD_DST);
76 
77  rv = opennsl_l2_addr_add(DEFAULT_UNIT, &addr);
78 
79  return rv;
80 }

91 int example_l2_firewall_allow_station(opennsl_mac_t mac, opennsl_vlan_t vlan)
92 {
93  opennsl_error_t rv;
94  opennsl_l2_addr_t addr;
95 
96  /* Check whether the station exists in the firewall rules */
97  rv = opennsl_l2_addr_get(DEFAULT_UNIT, mac, vlan, &addr);
98 
99  if (rv == OPENNSL_E_NONE) {
100  rv = opennsl_l2_addr_delete(DEFAULT_UNIT, mac, vlan);
101  }
102 
103  return rv;
104 }

146 int example_read_user_choice(int *choice)
147 {
148  char val;
149  char digits[MAX_DIGITS_IN_CHOICE + 1];
150  int idx = 0;
151  int valid = TRUE;
152 
153  /* parse input string until \n */
154  while((val = getchar()) != '\n')
155  {
156  if ((val >= '0' && val <= '9') && idx < MAX_DIGITS_IN_CHOICE)
157  {
158  digits[idx++] = val;
159  }
160  else
161  {
162  valid = FALSE;
163  }
164  }
165  if ((valid == TRUE) && idx != 0)
166  {
167  digits[idx] = '\0';
168  *choice = atoi(digits);
169  return(OPENNSL_E_NONE);
170  }
171  else
172  {
173  *choice = -1;
174  return(OPENNSL_E_FAIL);
175  }
176 }

186 int main(int argc, char *argv[])
187 {
188  opennsl_error_t rv;
189  opennsl_vlan_t vlan = 1;
190  int choice;
191 
192  /* Example MAC address of stations A, B, C and D */
193  opennsl_mac_t station_a = {0x00, 0x00, 0x00, 0x00, 0x00, 0x1};
194  opennsl_mac_t station_d = {0x00, 0x00, 0x00, 0x00, 0x00, 0x4};
195 
196  if((argc != 1) || ((argc > 1) && (strcmp(argv[1], "--help") == 0))) {
197  printf("%s\n\r", example_usage);
198  return OPENNSL_E_PARAM;
199  }
200 
201  /* Initialize the system. */
202  printf("Initializing the system.\r\n");
203  rv = opennsl_driver_init();
204 
205  if(rv != OPENNSL_E_NONE) {
206  printf("\r\nFailed to initialize the system. Error %s\r\n",
207  opennsl_errmsg(rv));
208  return rv;
209  }
210 
211  /* Add ports to default vlan. */
212  printf("Adding ports to default vlan.\r\n");
213  rv = example_switch_default_vlan_config(0);
214  if(rv != OPENNSL_E_NONE) {
215  printf("\r\nFailed to add default ports. rv: %d\r\n", rv);
216  return rv;
217  }
218 
219  while (1) {
220  printf("\r\nUser menu: Select one of the following options\r\n");
221  printf("1. Apply firewall\n");
222  printf("2. Remove firewall\n");
223  printf("9. Launch diagnostic shell\n");
224  printf("0. Quit the application\n");
225 
226  if(example_read_user_choice(&choice) != OPENNSL_E_NONE)
227  {
228  printf("Invalid option entered. Please re-enter.\n");
229  continue;
230  }
231  switch(choice){
232 
233  case 1:
234  {
235  /* Apply firewall rules to block station A and D */
236  rv = example_l2_firewall_block_station(station_a, vlan);
237  if (rv == OPENNSL_E_NONE) {
238  printf("L2 firewall: station A blocked successfully\n");
239  } else {
240  printf("L2 firewall: failed to apply rule. Error %s\n",
241  opennsl_errmsg(rv));
242  return rv;
243  }
244 
245  rv = example_l2_firewall_block_station(station_d, vlan);
246  if (rv == OPENNSL_E_NONE) {
247  printf("L2 firewall rule: station D blocked successfully\n");
248  } else {
249  printf("L2 firewall: failed to apply rule. Error %s\n",
250  opennsl_errmsg(rv));
251  return rv;
252  }
253  break;
254  } /* End of case 1 */
255  case 2:
256  {
257  /* Remove the firewall rules applied to station A and D */
258  rv = example_l2_firewall_allow_station(station_a, vlan);
259  if ((rv == OPENNSL_E_NONE) || (rv == OPENNSL_E_NOT_FOUND)) {
260  printf("L2 firewall: station A un-blocked successfully\n");
261  } else {
262  printf("L2 firewall: failed to apply rule. Error %s\n",
263  opennsl_errmsg(rv));
264  return rv;
265  }
266 
267  rv = example_l2_firewall_allow_station(station_d, vlan);
268  if ((rv == OPENNSL_E_NONE) || (rv == OPENNSL_E_NOT_FOUND)) {
269  printf("L2 firewall rule: station D un-blocked successfully\n");
270  } else {
271  printf("L2 firewall: failed to apply rule. Error %s\n",
272  opennsl_errmsg(rv));
273  return rv;
274  }
275  break;
276  } /* End of case 2 */
277  case 9:
278  {
279  opennsl_driver_shell();
280  break;
281  }
282  case 0:
283  {
284  printf("Exiting the application.\n");
285  return OPENNSL_E_NONE;
286  }
287  default:
288  break;
289  } /* End of switch */
290  } /* End of while */
291 
292  return OPENNSL_E_NONE;
293 }
